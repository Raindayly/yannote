用暴力双指针解了 时间复杂度很高但还是通过了
正确的解法是滑动窗口
我的解法
``` ts
function lengthOfLongestSubstring(s: string): number {
    if(s.length<2) {
        return s.length
    }
    let p1 = 0
    let p2 = 0 
    let sArr = Array.from(s)
    let sL = sArr.length
    let maxLen = 0
    while(p1 < sL) {

        let hasRepeat = false
        let repeatArr:string[] = []
        while(p2 <= sL) {
            hasRepeat = !!repeatArr.find(i=>i === sArr[p2]) 
            if(hasRepeat || p2 === sL){
                maxLen = Math.max(maxLen,repeatArr.length);
                break
            }

            repeatArr.push(sArr[p2])

            p2++
        }
        p1++
        p2 = p1
        if(maxLen > (sL - p1)) break
    }
    return maxLen;
};

```

``` javascript
function lengthOfLongestSubstring(s: string): number {
    let left = 0
    let right = 0
    let map = new Map()
    let maxL = 0
    while(right < s.length) {
        const currentChar = s[right];
        let has =  map.has(currentChar)
        let index = map.get(currentChar)
        if(has && index >= left) {
            left = map.get(currentChar)! + 1; 
        }
        maxL = Math.max(right - left + 1,maxL)

        map.set(s[right],right)
        right++

       
    }
    return maxL
};
```
